%Establece los rangos de la grafica
axis ([-1.2 6.5 -1.2 1.2])
grid on
hold on
%Configuración de la grafica
xlabel('x', 'Color', 'black');
ylabel('y', 'Color', 'black');
title('Función por partes', 'Color', 'black');
%Creación de las funciones
%Funcion 1
x1 = [-1 0];
y1 = [0 0];
%Funcion 2
x2 = [0 1];
y2 = [-1 -1];
%Funcion 3
x3 = 1:0.01:2;
y3 = sin(-pi*x3);
%Funcion 4
x4 = 2:0.01:4;
y4 = sin(pi*x4);
%Funcion 5
x5 = [4 6.23];
y5 = [0 0];
%Union de las funciones
x = [x3 x4 x5];
y = [y3 y4 y5];
%Grafica
plot(x, y, x1, y1, x2, y2, 'b', 'Linewidth', 2)
